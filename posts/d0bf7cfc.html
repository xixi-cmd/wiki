获取用户输入

import java.util.Scanner;

public class scanner {
    public static void main(String[] args) {
        System.out.println("enter two numbers:");

        Scanner in = new Scanner(System.in);
        int num = in.nextInt();
        int doub = in.nextInt();
        int resu = num * doub;

        System.out.println("result is " + num + "x" + doub + "=" + resu);

        in.close();
    }
}
